149    def process_callback(self, data) -> None:
150        """
151        Verwerkt binnenkomende data van de robot en routeert deze naar het juiste onderdeel.
152
153        Args:
154            data (bytes): De ontvangen data inclusief header, payload en CRC.
155        """
156        response = data[1]
157
158        if response == (SaraRobotCommands.CMD_VERSION_BODY | SaraRobotCommands.RESP_BIT):
159            self.body.new_version_data(data)
160            return
161
162        if response == (SaraRobotCommands.CMD_VERSION_HEAD | SaraRobotCommands.RESP_BIT):
163            self.head.new_version_data(data)
164            return
165
166        if response == (SaraRobotCommands.CMD_GET_BATTERY | SaraRobotCommands.RESP_BIT):
167            self.body.battery.new_data(data)
168            if self.logging:
169                self.body.battery.print_state()
170            return
171
172        if response == (
173            SaraRobotCommands.CMD_GET_DISTANCESENSORS | SaraRobotCommands.RESP_BIT
174        ):
175            self.body.distance_sensors.new_data(data)
176            if self.logging:
177                self.body.distance_sensors.print_values()
178            return
179
180        if response == (SaraRobotCommands.CMD_GET_COMPASS | SaraRobotCommands.RESP_BIT):
181            self.body.compass.new_data(data)
182            if self.logging:
183                self.body.compass.print_values()
184            return
185
186        if response == (SaraRobotCommands.CMD_COMP_MOVE | SaraRobotCommands.RESP_BIT):
187            self.body.compass.rotate_absolute_ready(data)
188            return
189
190        if response == (
191            SaraRobotCommands.CMD_GET_ENCODERS | SaraRobotCommands.RESP_BIT
192        ):
193            self.body.encoders.new_data(data)
194            if self.logging:
195                self.body.encoders.print_values()
196            return
197
198        if response == (
199            SaraRobotCommands.CMD_GET_MOTIONSENSORS | SaraRobotCommands.RESP_BIT
200        ):
201            self.body.motion_sensors.new_data(data)
202            if self.logging:
203                self.body.motion_sensors.print_values()
204            return
205
206
207        if response == (
208            SaraRobotCommands.CMD_HEAD_TOUCHSENSORS | SaraRobotCommands.RESP_BIT
209        ):
210            self.head.touch_sensors.new_data(data)
211            if self.logging:
212                self.head.touch_sensors.print_values()
213            return
214
215        if response == (
216            SaraRobotCommands.CMD_BODY_TOUCHSENSORS | SaraRobotCommands.RESP_BIT
217        ):
218            self.body.touch_sensors.new_data(data)
219            if self.logging:
220                self.body.touch_sensors.print_values()
221            return
222
223        # If not decoded, print the data
224        hex_values = " ".join([format(x, "02X") for x in data])
225        print("< NOT DECODED: " + hex_values)
